Integrations tab handleSave:

```typescript
const handleIntegrationsSave = async (e: FormEvent) => {
  e.preventDefault()
  setIntegrationsLoading(true)
  const updates: Record<string, unknown> = {
    unsubscribe_footer_text: unsubscribeFooter || null,
    updated_at: new Date().toISOString(),
  }
  if (apiKeyDirty) {
    updates.resend_api_key = apiKey || null
  }
  const { error } = await supabase
    .from('profiles')
    .update(updates)
    .eq('id', user?.id ?? '')
  setIntegrationsLoading(false)
  if (error) {
    showToast(error.message, 'error')
  } else {
    showToast('Saved successfully.')
    if (apiKeyDirty) {
      setApiKeyExists(!!apiKey)
      setApiKey('')
      setApiKeyDirty(false)
    }
    await refreshProfile()
  }
}
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-10 | Tampering | profiles UPDATE via settings | mitigate | Existing RLS on profiles: `auth.uid() = id` scopes update to own profile only |
| T-07-11 | Information Disclosure | resend_api_key in network | accept | Sent via HTTPS in Supabase client request body. No logging in codebase (CONVENTIONS.md). Key stored as plaintext in DB -- acceptable for MVP per RESEARCH.md A3 |
| T-07-12 | Information Disclosure | resend_api_key in UI | mitigate | Input type="password" masks display. Value never populated into input field -- only dirty state is sent. apiKeyDirty guard prevents overwriting with empty string on unrelated tab save |
| T-07-13 | Denial of Service | Repeated settings saves | accept | Low risk -- single user per workspace, standard Supabase rate limits apply |
</threat_model>
