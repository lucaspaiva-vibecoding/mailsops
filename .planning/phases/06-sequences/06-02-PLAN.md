```typescript
interface StepEditorPanelProps {
  step: {
    localId: string         // stable UUID for React key (NOT DB id)
    delayDays: number
    subject: string
    bodyHtml: string
    bodyJson: Record<string, unknown> | null
  }
  stepIndex: number        // 0-based index for display as "Step {stepIndex+1}"
  isFirst: boolean
  isLast: boolean
  isOnly: boolean          // hide remove button if only 1 step

  onChange: (updated: StepEditorPanelProps['step']) => void

}

const editor = useEditor({

    StarterKit,
    Link.configure({ openOnClick: false }),

      ...step,
      bodyHtml: editor.getHTML(),

  },
})
```
